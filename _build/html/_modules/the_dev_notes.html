def to_run_things() -> None:
    """
    Go to Linux terminal. Make sure that when you typed ls, 
    you can see the level files and the .py files like game.py
    and others.

    1. To run terminal version of the game:
    run 'python3 mp1.py <level>'

    2. To run the game with the GUI:
    run 'python3 gui.py'

    3. To run the test:
    run 'pytest test.py'

    4. To run Pyright:
    run 'pyright <file>'

    5. To run mypy:
    run 'mypy <file> --strict'

    More details are on the README file.
    """
    return
